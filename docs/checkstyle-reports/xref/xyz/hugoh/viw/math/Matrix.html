23      public static float[][] mul(float[][] a, float[][] b) {
24          int aRows = a[0].length;
25          int bRows = b[0].length;
26          int aCols = a.length;
27          int bCols = b.length;
28  
29          float[][] newMatrix = new float[bCols][aRows];
30  
31          if (aCols != bRows) {
32              throw new IllegalArgumentException("Matrix a's column count doesn't match matrix b's row count");
33          }
34  
35          for (int i = 0; i < aRows; ++i) {
36              for (int j = 0; j < aCols; ++j) {
37                  for (int k = 0; k < bCols; ++k) {
38                      newMatrix[k][i] += a[j][i] * b[k][j];
39                  }
40              }
41          }
42          return newMatrix;
43      }

45      public static float[][] translate4x4(float[][] matrix, float[] vector) {
46          float[][] newMatrix = new float[4][4];
47          System.arraycopy(matrix, 0, newMatrix, 0, matrix.length);
48          float[] a = Array.mulFloat(matrix[0], vector[0]);
49          float[] b = Array.mulFloat(matrix[1], vector[1]);
50          float[] c = Array.mulFloat(matrix[2], vector[2]);
51          float[] sum = Array.sum(a, b);
52          sum = Array.sum(sum, c);
53          sum = Array.sum(sum, matrix[3]);
54          newMatrix[3] = sum;
55          return newMatrix;
56      }
