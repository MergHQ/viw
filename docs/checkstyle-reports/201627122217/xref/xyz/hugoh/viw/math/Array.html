6   public class Array {
7       public Array() {}

9       public static float[] mul(float[] a, float[] b) {
10          if (a.length != b.length) {
11              throw new IllegalArgumentException("Array sizes do not match");
12          }
13          float[] res = new float[a.length];
14          for (int i = 0; i < a.length; ++i) {
15              res[i] = a[i] * b[i];
16          }
17          return res;
18      }
19  
20      public static float[] mulFloat(float[] a, float b) {
21          float[] res = new float[a.length];
22          for (int i = 0; i < a.length; ++i) {
23              res[i] = a[i] * b;
24          }
25          return res;
26      }
27  
28      public static float[] sum(float[] a, float[] b) {
29          if (a.length != b.length) {
30              throw new IllegalArgumentException("Array sizes do not match");
31          }
32          float[] res = new float[a.length];
33          for (int i = 0; i < a.length; ++i) {
34              res[i] = b[i] + a[i];
35          }
36          return res;
37      }
