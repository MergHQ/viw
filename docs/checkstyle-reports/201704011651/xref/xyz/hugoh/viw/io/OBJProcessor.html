24  public class OBJProcessor {
25      private List<Float> vertices;
26      private List<Float> normals;
27      private List<Integer> indices;

29      public OBJProcessor() {
30          vertices = new ArrayList<>();
31          normals = new ArrayList<>();
32          indices = new ArrayList<>();
33      }

35      public Mesh load3DObject(String file) {
36          Mesh mesh = new Mesh("");
37          try {
38              Stream<String> stream = Files.lines(Paths.get(file));
39              stream.forEach(line -> processLine(line, mesh));
40          } catch (IOException e) {
41              System.out.println(e.getMessage());
42          }
43          createGLHandles(mesh);
44          return mesh;
45      }

47      private void createGLHandles(Mesh m) {
48          // I can't believe there isn't any good way to do this.
49          double[] vertexArray = vertices.stream().mapToDouble(f -> f).toArray();
50          double[] normalArray = normals.stream().mapToDouble(f -> f).toArray();
51          int[] indexArray = indices.stream().mapToInt(i -> i).toArray();
52          int vao = GL30.glGenVertexArrays();
53          GL30.glBindVertexArray(vao);
54  
55          // Vertices
56          GL20.glEnableVertexAttribArray(0);
57          int vbo = GL15.glGenBuffers();
58          GL15.glBindBuffer(GL15.GL_ARRAY_BUFFER, vbo);
59          DoubleBuffer vertexData = BufferUtils.createDoubleBuffer(vertexArray.length * 3);
60          vertexData.put(vertexArray);
61          vertexData.flip();
62          GL15.glBufferData(GL15.GL_ARRAY_BUFFER, vertexData, GL15.GL_STATIC_DRAW);
63          GL20.glVertexAttribPointer(0, 3,GL11.GL_DOUBLE, false, GL11.GL_FALSE, 32);
64  
65          // Normals
66          GL20.glEnableVertexAttribArray(1);
67          GL15.glGenBuffers();
68          int nbo = GL15.glGenBuffers();
69          GL15.glBindBuffer(GL15.GL_ARRAY_BUFFER, nbo);
70          DoubleBuffer normalData = BufferUtils.createDoubleBuffer(normalArray.length * 3);
71          vertexData.put(normalArray);
72          vertexData.flip();
73          GL15.glBufferData(GL15.GL_ARRAY_BUFFER, normalData, GL15.GL_STATIC_DRAW);
74          GL20.glVertexAttribPointer(0, 3,GL11.GL_DOUBLE, false, GL11.GL_FALSE, 32);
75  
76          m.setVertexArrayObject(vao);
77          GL30.glBindVertexArray(0);
78      }

80      private void processLine(String line, Mesh m) {
81          String[] lineArgs = line.split(" ");
82          if (lineArgs.length != 0 && !lineArgs[0].equals("#")) {
83              switch (lineArgs[0]) {
84                  case "g":
85                      m.setName(lineArgs[1]);
86                      break;
87                  case "v":
88                      try {
89                          vertices.add(Float.parseFloat(lineArgs[2]));
90                          vertices.add(Float.parseFloat(lineArgs[4]));
91                          vertices.add(Float.parseFloat(lineArgs[6]));
92                      } catch (Exception e) {
93                          System.out.println(e.getMessage());
94                      }
95                      break;
96                  case "vn":
97                      try {
98                          normals.add(Float.parseFloat(lineArgs[2]));
99                          normals.add(Float.parseFloat(lineArgs[4]));
100                         normals.add(Float.parseFloat(lineArgs[lineArgs.length == 7 ? 6 : 5]));
101                     } catch (Exception e) {
102                         System.out.println(e.getMessage());
103                     }
104                     break;
105                 case "f":
106                     // index
107                     break;
108             }
109         }
110     }

112     private static <T> void fromListToArray(ArrayList<?> input, T[] out) {
113         if (out.length != input.size()) {
114             throw new IllegalArgumentException("Array sizes do no match");
115         }
116         for (int i = 0; i < input.size(); ++i) {
117             out[i] = (T) input.get(i);
118         }
119     }
