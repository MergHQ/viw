11  public class Scene {
12      private List<Mesh> meshList;
13      private Shader shader;
14  
15      public Scene() {
16          meshList = new ArrayList<>();
17      }
18  
19      public void addMesh(Mesh m) {
20          meshList.add(m);
21      }
22  
23      public Mesh getMesh(String name) {
24          return meshList.stream()
25                  .filter(m -> m.getName().equals(name))
26                  .limit(1)
27                  .collect(Collectors.toList())
28                  .get(0);
29      }
30  
31      public List<Mesh> getMeshList() {
32          return meshList;
33      }
34  }
